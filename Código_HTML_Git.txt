    <script>
        // Função para buscar os dados mais recentes da API
        async function fetchLatestData() {
            try {
                // Faremos uma chamada GET para um novo endpoint da nossa API
                const response = await fetch('/latest'); 
                const data = await response.json();
                
                // Atualiza os elementos HTML com os dados
                document.getElementById('luminosidade').textContent = data.luminosidade ? data.luminosidade.toFixed(2) : '--';
                document.getElementById('temperatura').textContent = data.temperatura ? data.temperatura.toFixed(2) : '--';
                document.getElementById('umidade').textContent = data.umidade ? data.umidade.toFixed(2) : '--';
                document.getElementById('nivel_ruido').textContent = data.nivel_ruido ? data.nivel_ruido : '--';
                document.getElementById('direcao_movimento').textContent = data.direcao_movimento || '--';
                
                // Atualiza o horário da última atualização
                const now = new Date();
                const timeString = now.toLocaleTimeString('pt-BR');
                document.getElementById('last-update').textContent = `Última atualização: ${timeString}`;
                
            } catch (error) {
                console.error('Erro ao buscar dados:', error);
                document.getElementById('last-update').textContent = 'Erro ao buscar dados!';
            }
        }

        // Busca os dados imediatamente
        fetchLatestData();
        
        // Configura para buscar dados a cada 5 segundos
        setInterval(fetchLatestData, 5000); 
    </script>
